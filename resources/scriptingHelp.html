Example Usage of the Scripting Console

Example 1: hide all ensemble memberships 
	for each (var el in memberships){
		el.setVisible(false)
	}
Example 2: show only ensemble memberships with coordinator V7 
	for each (var el in memberships){
		if (el.getCoordinator().equals("V7")){
			el.setVisible(true)
		} else {
			el.setVisible(false)		
		}
	}
Example 3: print info about ensemble memberships 
	var count = 0;
	for each (var el in memberships){
		print("Item " + count + ": ")
		print("Ensemble name=" + el.getEnsembleName() + ", ")
		print("Coordinator=" + el.getCoordinator() + ", ")
		print("Member=" + el.getMember())
		print()
		count += 1
	}
Example 4: set image representing all people 
	general.setPersonImage("C:/image.png")
Example 5: set image representing persons V7, V11 
	general.setPersonImage("C:/image.png", "V7", "V11")
Example 6: set image representing persons V7, V11 - variant II 
	var jsArr = ["V7","V11"]
	var strArrType = Java.type("java.lang.String[]")
	var javaArr = Java.to(jsArr, strArrType)
	general.setPersonImage("C:/image.png", javaArr)
Example 7: set image representing persons V7,V11 back to the original circles 
	general.setPersonImage(null, "V7", "V11")
Example 8: set color of the ensemble memberships with coordinator V7 to red, others to green 
	for each (var el in memberships){
		if (el.getCoordinator().equals("V7")){
			el.setColor(Color.RED)
		} else {
			el.setColor(Color.GREEN)
		}
	}
